  <div class="flip-card" id="flipCard">
    <div class="front" id="frontFace">
      <div class="day" id="frontDay">Mon</div>
      <div class="date" id="frontDate">23</div>
    </div>
    <div class="back" id="backFace">
      <div class="day" id="backDay">Tue</div>

<script>
const flipCard = document.getElementById("flipCard");
const frontDay = document.getElementById("frontDay");
const frontDate = document.getElementById("frontDate");
const backDay = document.getElementById("backDay");
const backDate = document.getElementById("backDate");

const days = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];

function updateFront() {
  const now = new Date();
  frontDay.textContent = days[now.getDay()];
  frontDate.textContent = String(now.getDate()).padStart(2, "0");
}

function updateBack() {
  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 1);
  backDay.textContent = days[tomorrow.getDay()];
  backDate.textContent = String(tomorrow.getDate()).padStart(2, "0");
}

function flipAtMidnight() {
  const now = new Date();
  const msToMidnight =
    new Date(now.getFullYear(), now.getMonth(), now.getDate() + 1, 0, 0, 0) - now;

  setTimeout(() => {
    flipCard.classList.add("flipped");

    setTimeout(() => {
      updateFront();      // 앞면을 새로운 날짜로 갱신
      updateBack();       // 다시 다음 날 준비
      flipCard.classList.remove("flipped");
      flipAtMidnight();   // 다음 자정 예약
    }, 1000); // 플립 애니메이션 후 업데이트
  }, msToMidnight);
}

// 초기 설정
updateFront();
updateBack();
flipAtMidnight();
</script>
